```ts
  return { from: sel.from, to: sel.to, anchor: sel.anchor, head: sel.head };

private filesFromDataTransfer(dataTransfer: DataTransfer | null): File[] {

private insertFileHandlerMarkdown(markdown: string | string[], from: number, to: number): void {
  const text = Array.isArray(markdown) ? markdown.join('\n') : markdown;
  if (!text) return;
  this.view.dispatch({
    changes: { from, to, insert: text },
    selection: { anchor: from + text.length },
    scrollIntoView: true,
  });
}
```

- [ ] **Step 3: Add async handler path**

Add:

```ts
private async handleFiles(
  files: File[],
  source: GalleyFileSource,
  event: ClipboardEvent | DragEvent,
  from = this.view.state.selection.main.from,
  to = this.view.state.selection.main.to,
): Promise<void> {
  const handler = this.callbacks.onFiles;
  if (!handler || files.length === 0) return;

  const input = {
    files,
    source,
    event,
    view: this.view,
    selection: this.getSelectionInfo(),
  };

  try {
    const result = await handler(input);
    if (result === false || result === null) return;
    this.insertFileHandlerMarkdown(result, from, to);
  } catch (error) {
    this.callbacks.onFileError?.(error, input);
    if (!this.callbacks.onFileError) {
      console.error('Galley file handler failed', error);
    }
  }
}
```

- [ ] **Step 4: Wire paste/drop DOM handlers**

Change the existing `paste` handler to extract files first:

```ts
paste: (e, view) => {
  const files = this.filesFromDataTransfer(e.clipboardData);
  if (files.length > 0 && this.callbacks.onFiles) {
    e.preventDefault();
    void this.handleFiles(files, 'paste', e);
  }
  this.callbacks.onPaste?.(e, view);
},
dragover: (e) => {
  if (this.callbacks.onFiles && this.filesFromDataTransfer(e.dataTransfer).length > 0) {
    e.preventDefault();
  }
},
drop: (e) => {
  const files = this.filesFromDataTransfer(e.dataTransfer);
  if (!this.callbacks.onFiles || files.length === 0) return;
  e.preventDefault();
  const pos = this.view.posAtCoords({ x: e.clientX, y: e.clientY });
  const insertAt = pos ?? this.view.state.selection.main.from;
  void this.handleFiles(files, 'drop', e, insertAt, insertAt);
},
```

- [ ] **Step 1: Create parser and serializer**

Create `src/image-markdown.ts`:

```ts
import type { EditorState } from '@codemirror/state';
import type { GalleyImageInfo, GalleyImageMetadataInput } from './types';

const IMAGE_RE = /^!\[(?<alt>[^\]]*)\]\((?<target>.*?)\)(?<attrs>\{[^}]*\})?$/;

function parseAttrs(attrs: string | undefined): { width?: number; height?: number; rest: string[] } {
  const rest: string[] = [];
  const result: { width?: number; height?: number; rest: string[] } = { rest };
  if (!attrs) return result;
  for (const part of attrs.slice(1, -1).trim().split(/\s+/).filter(Boolean)) {
    const [key, value] = part.split('=');
    const numeric = Number(value);
    if (key === 'width' && Number.isFinite(numeric)) result.width = numeric;
    else if (key === 'height' && Number.isFinite(numeric)) result.height = numeric;
    else rest.push(part);
  }
  return result;
}

export function parseImageMarkdown(raw: string, from = 0, to = raw.length): GalleyImageInfo | null {
  const match = IMAGE_RE.exec(raw.trim());
  if (!match?.groups) return null;
  const titleMatch = /^(?<url>.+?)\s+"(?<title>[^"]*)"$/.exec(match.groups.target.trim());
  const attrs = parseAttrs(match.groups.attrs);
  return {
    alt: match.groups.alt,
    url: titleMatch?.groups?.url ?? match.groups.target.trim(),
    ...(titleMatch?.groups?.title ? { title: titleMatch.groups.title } : {}),
    ...(attrs.width ? { width: attrs.width } : {}),
    ...(attrs.height ? { height: attrs.height } : {}),
    raw,
    from,
    to,
  };
}

export function serializeImageMarkdown(image: GalleyImageInfo, patch: GalleyImageMetadataInput = {}): string {
  const next = { ...image, ...patch };
  const title = next.title ? ` "${next.title}"` : '';
  const attrs = [
    next.width ? `width=${next.width}` : null,
    next.height ? `height=${next.height}` : null,
  ].filter(Boolean).join(' ');
  return `![${next.alt}](${next.url}${title})${attrs ? `{${attrs}}` : ''}`;
}

export function imageAtSelection(state: EditorState): GalleyImageInfo | null {
  const sel = state.selection.main;
  const line = state.doc.lineAt(sel.from);
  const match = /!\[[^\]]*\]\(.*?\)(?:\{[^}]*\})?/.exec(line.text);
  if (!match) return null;
  const from = line.from + match.index;
  const to = from + match[0].length;
  if (sel.from < from || sel.from > to) return null;
  return parseImageMarkdown(match[0], from, to);
}
```

```ts
import type { EditorView } from '@codemirror/view';
import type { GalleyImageMetadataInput } from '../types';
import { imageAtSelection, serializeImageMarkdown } from '../image-markdown';

export function updateImageMetadata(view: EditorView, input: GalleyImageMetadataInput): boolean {
  const image = imageAtSelection(view.state);
  if (!image) return false;
  const next = serializeImageMarkdown(image, input);
  view.dispatch({
    changes: { from: image.from, to: image.to, insert: next },
    selection: { anchor: image.from + next.length },
    scrollIntoView: true,
  });
  return true;
}

export function clearImageDimensions(view: EditorView): boolean {
  return updateImageMetadata(view, { width: null, height: null });
}
```
