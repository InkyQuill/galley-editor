```ts
import { syntaxTree } from '@codemirror/language';
import type { EditorState } from '@codemirror/state';

export type TableAlignment = 'left' | 'center' | 'right' | null;
export type TableDirection = 'left' | 'right' | 'up' | 'down';

export interface TableCellRef {
  row: number;
  column: number;
}

export interface GalleyTableCell extends TableCellRef {
  text: string;
  sourceFrom: number;
  sourceTo: number;
  header: boolean;
}

export interface GalleyTable {
  from: number;
  to: number;
  rows: GalleyTableCell[][];
  alignments: TableAlignment[];
  columnCount: number;
}

export function parseMarkdownTable(source: string, from = 0): GalleyTable | null {
  if (source.includes('\\|')) return null;
  const lineStarts = source.split('\n').reduce<number[]>((starts, line, index, lines) => {
    if (index === 0) return starts;
    const previous = lines.slice(0, index).join('\n');
    starts.push(previous.length + 1);
    return starts;
  }, [0]);
  const lines = source.split('\n').filter((line) => line.trim().length > 0);
  if (lines.length < 2) return null;

  const header = parseRow(lines[0], from + lineStarts[0], 0, true);
  const separator = splitRow(lines[1]);
  if (header.length === 0 || separator.length === 0) return null;
  if (!separator.every((cell) => /^:?-{1,}:?$/.test(cell.trim()))) return null;

  const columnCount = header.length;
  const rows = [padCells(header, columnCount)];
  for (let index = 2; index < lines.length; index += 1) {
    rows.push(padCells(parseRow(lines[index], from + lineStarts[index], rows.length, false), columnCount));
  }

  return {
    from,
    to: from + source.length,
    rows,
    alignments: Array.from({ length: columnCount }, (_value, index) => parseAlignment(separator[index] ?? '')),
    columnCount,
  };
}

export function serializeMarkdownTable(table: GalleyTable): string {
  const header = serializeRow(table.rows[0] ?? []);
  const separator = serializeRow(table.alignments.map(alignmentCell));
  const body = table.rows.slice(1).map(serializeRow);
  return `${[header, separator, ...body].join('\n')}\n`;
}

export function tableAtSelection(state: EditorState): GalleyTable | null {
  const pos = state.selection.main.head;
  return tableAtPosition(state, pos);
}

export function tablesAtSelections(state: EditorState): GalleyTable[] {
  const byRange = new Map<string, GalleyTable>();
  for (const range of state.selection.ranges) {
    const table = tableAtPosition(state, range.head);
    if (table) byRange.set(`${table.from}:${table.to}`, table);
  }
  return [...byRange.values()].sort((a, b) => a.from - b.from);
}

function tableAtPosition(state: EditorState, pos: number): GalleyTable | null {
  let found: GalleyTable | null = null;
  syntaxTree(state).iterate({
    enter(node) {
      if (found || node.name !== 'Table') return;
      if (pos < node.from || pos > node.to) return;
      found = parseMarkdownTable(state.sliceDoc(node.from, node.to), node.from);
    },
  });
  return found;
}

export function tableCellAtPosition(table: GalleyTable, pos: number): GalleyTableCell {
  for (const row of table.rows) {
    for (const cell of row) {
      if (pos >= cell.sourceFrom && pos <= cell.sourceTo) return cell;
    }
  }
  return table.rows[0]?.[0] ?? emptyCell(0, 0, table.from, true);
}

export function tableCell(table: GalleyTable, ref: TableCellRef): GalleyTableCell | null {
  return table.rows[ref.row]?.[ref.column] ?? null;
}

export function updateTableCell(table: GalleyTable, ref: TableCellRef, text: string): GalleyTable {
  return {
    ...table,
    rows: table.rows.map((row, rowIndex) =>
      row.map((cell, columnIndex) =>
        rowIndex === ref.row && columnIndex === ref.column ? { ...cell, text } : cell,
      ),
    ),
  };
}

export function tableNavigationCell(table: GalleyTable, ref: TableCellRef, direction: TableDirection): TableCellRef {
  const lastRow = Math.max(0, table.rows.length - 1);
  const lastColumn = Math.max(0, table.columnCount - 1);
  if (direction === 'left') return { row: ref.row, column: Math.max(0, ref.column - 1) };
  if (direction === 'right') return { row: ref.row, column: Math.min(lastColumn, ref.column + 1) };
  if (direction === 'up') return { row: Math.max(0, ref.row - 1), column: ref.column };
  return { row: Math.min(lastRow, ref.row + 1), column: ref.column };
}

export function cellKey(ref: TableCellRef | null): string {
  return ref ? `${ref.row}:${ref.column}` : '';
}

function parseRow(line: string, lineFrom: number, row: number, header: boolean): GalleyTableCell[] {
  const trimmedStart = line.match(/^\s*/)?.[0].length ?? 0;
  const hasLeadingPipe = line.trimStart().startsWith('|');
  const contentStart = trimmedStart + (hasLeadingPipe ? 1 : 0);
  const trimmed = line.trim().replace(/^\|/, '').replace(/\|$/, '');
  let cursor = lineFrom + contentStart;
  return trimmed.split('|').map((raw, column) => {
    const leading = raw.match(/^\s*/)?.[0].length ?? 0;
    const trailing = raw.match(/\s*$/)?.[0].length ?? 0;
    const text = raw.trim();
    const sourceFrom = cursor + leading;
    const sourceTo = cursor + raw.length - trailing;
    cursor += raw.length + 1;
    return { row, column, text, sourceFrom, sourceTo, header };
  });
}

function splitRow(line: string): string[] {
  return line.trim().replace(/^\|/, '').replace(/\|$/, '').split('|').map((cell) => cell.trim());
}

function parseAlignment(cell: string): TableAlignment {
  const normalized = cell.trim();
  const left = normalized.startsWith(':');
  const right = normalized.endsWith(':');
  if (left && right) return 'center';
  if (right) return 'right';
  if (left) return 'left';
  return null;
}

function alignmentCell(alignment: TableAlignment): string {
  if (alignment === 'left') return ':---';
  if (alignment === 'center') return ':---:';
  if (alignment === 'right') return '---:';
  return '---';
}

function serializeRow(row: Array<GalleyTableCell | string>): string {
  return `| ${row.map((cell) => typeof cell === 'string' ? cell : cell.text).join(' | ')} |`;
}

function padCells(cells: GalleyTableCell[], columnCount: number): GalleyTableCell[] {
  const next = [...cells];
  while (next.length < columnCount) next.push(emptyCell(cells[0]?.row ?? 0, next.length, cells.at(-1)?.sourceTo ?? 0, cells[0]?.header ?? false));
  return next.slice(0, columnCount).map((cell, column) => ({ ...cell, column }));
}

function emptyCell(row: number, column: number, pos: number, header: boolean): GalleyTableCell {
  return { row, column, text: '', sourceFrom: pos, sourceTo: pos, header };
}
```

```ts
import { EditorSelection, type TransactionSpec } from '@codemirror/state';
import type { EditorView } from '@codemirror/view';
import type { CommandFn } from '../types';
import {
  serializeMarkdownTable,
  tableAtSelection,
  tableCell,
  tableCellAtPosition,
  tablesAtSelections,
  updateTableCell,
  type GalleyTable,
  type TableAlignment,
  type TableCellRef,
} from '../table-markdown';

export function replaceTable(view: EditorView, table: GalleyTable, spec: Partial<TransactionSpec> = {}): boolean {
  const text = serializeMarkdownTable(table);
  view.dispatch({
    changes: { from: table.from, to: table.to, insert: text },
    ...spec,
  });
  return true;
}

export function replaceTables(view: EditorView, tables: GalleyTable[]): boolean {
  if (tables.length === 0) return false;
  const changes = tables
    .map((table) => ({ from: table.from, to: table.to, insert: serializeMarkdownTable(table) }))
    .sort((a, b) => a.from - b.from);
  view.dispatch({ changes });
  return true;
}

export const normalizeTable: CommandFn = (view) => {
  return replaceTables(view, tablesAtSelections(view.state));
};

export function commitTableCell(view: EditorView, ref: TableCellRef, text: string): boolean {
  const table = tableAtSelection(view.state);
  if (!table || !tableCell(table, ref)) return false;
  return replaceTable(view, updateTableCell(table, ref, text));
}

export const insertTableRowBefore: CommandFn = (view) => mutateRow(view, 'before');
export const insertTableRowAfter: CommandFn = (view) => mutateRow(view, 'after');
export const deleteTableRow: CommandFn = (view) => mutateRow(view, 'delete');
export const insertTableColumnBefore: CommandFn = (view) => mutateColumn(view, 'before');
export const insertTableColumnAfter: CommandFn = (view) => mutateColumn(view, 'after');
export const deleteTableColumn: CommandFn = (view) => mutateColumn(view, 'delete');

export const setTableColumnAlignment: CommandFn = (view, alignment) => {
  if (alignment !== 'left' && alignment !== 'center' && alignment !== 'right' && alignment !== null) return false;
  const table = tableAtSelection(view.state);
  if (!table) return false;
  const cell = tableCellAtPosition(table, view.state.selection.main.head);
  const alignments = [...table.alignments];
  alignments[cell.column] = alignment as TableAlignment;
  return replaceTable(view, { ...table, alignments });
};

export function revealTableSource(view: EditorView, ref?: TableCellRef): boolean {
  const table = tableAtSelection(view.state);
  if (!table) return false;
  const cell = ref ? tableCell(table, ref) : tableCellAtPosition(table, view.state.selection.main.head);
  if (!cell) return false;
  view.dispatch({ selection: EditorSelection.cursor(cell.sourceFrom) });
  return true;
}

function selectedCellForTable(view: EditorView, table: GalleyTable) {
  const range = view.state.selection.ranges.find((range) => range.head >= table.from && range.head <= table.to);
  return tableCellAtPosition(table, range?.head ?? table.from);
}

function mutateRow(view: EditorView, action: 'before' | 'after' | 'delete'): boolean {
  const tables = tablesAtSelections(view.state);
  if (tables.length === 0) return false;
  const changed: GalleyTable[] = [];
  for (const table of tables) {
    const cell = selectedCellForTable(view, table);
    const empty = Array.from({ length: table.columnCount }, (_value, column) => ({
      row: cell.row,
      column,
      text: '',
      sourceFrom: table.from,
      sourceTo: table.from,
      header: cell.row === 0,
    }));
    const rows = table.rows.map((row) => row.map((item) => ({ ...item })));
    if (action === 'delete') {
      if (cell.row === 0) continue;
      rows.splice(cell.row, 1);
    } else {
      rows.splice(action === 'before' ? cell.row : cell.row + 1, 0, empty);
    }
    changed.push(reindexRows({ ...table, rows }));
  }
  return replaceTables(view, changed);
}

function mutateColumn(view: EditorView, action: 'before' | 'after' | 'delete'): boolean {
  const tables = tablesAtSelections(view.state);
  if (tables.length === 0) return false;
  const changed: GalleyTable[] = [];
  for (const table of tables) {
    const cell = selectedCellForTable(view, table);
    if (action === 'delete' && table.columnCount <= 1) continue;
    const index = action === 'after' ? cell.column + 1 : cell.column;
    const rows = table.rows.map((row, rowIndex) => {
      const next = row.map((item) => ({ ...item }));
      if (action === 'delete') next.splice(cell.column, 1);
      else next.splice(index, 0, { row: rowIndex, column: index, text: '', sourceFrom: table.from, sourceTo: table.from, header: rowIndex === 0 });
      return next;
    });
    const alignments = [...table.alignments];
    if (action === 'delete') alignments.splice(cell.column, 1);
    else alignments.splice(index, 0, null);
    changed.push(reindexRows({ ...table, rows, alignments, columnCount: alignments.length }));
  }
  return replaceTables(view, changed);
}

function reindexRows(table: GalleyTable): GalleyTable {
  return {
    ...table,
    columnCount: table.alignments.length,
    rows: table.rows.map((row, rowIndex) =>
      row.map((cell, columnIndex) => ({ ...cell, row: rowIndex, column: columnIndex, header: rowIndex === 0 })),
    ),
  };
}
```

```ts
  tableTo: number;

const selectTableCell = StateEffect.define<SelectedTableCell | null>({
  map(value, changes) {
    if (!value) return null;
    const tableFrom = changes.mapPos(value.tableFrom, 1);
    const tableTo = changes.mapPos(value.tableTo, -1);
    return tableFrom < tableTo ? { ...value, tableFrom, tableTo } : null;
  },

const selectedTableCellField = StateField.define<SelectedTableCell | null>({
  create: () => null,
  update(value, transaction) {
    let next = value;
    if (next && transaction.docChanged) {
      const tableFrom = transaction.changes.mapPos(next.tableFrom, 1);
      const tableTo = transaction.changes.mapPos(next.tableTo, -1);
      next = tableFrom < tableTo ? { ...next, tableFrom, tableTo } : null;

    for (const effect of transaction.effects) {
      if (effect.is(selectTableCell)) {
        handled = true;
        next = effect.value;
```

```ts
class TableWidget extends WidgetType {
  table: GalleyTable;
  tableClass: string;
  selected: SelectedTableCell | null;
  canEdit: boolean;

  constructor(
    table: GalleyTable,
    tableClass: string,
    selected: SelectedTableCell | null,
    canEdit: boolean,
  ) {
    super();
    this.table = table;
    this.tableClass = tableClass;
    this.selected = selected;
    this.canEdit = canEdit;
  }

  eq(other: TableWidget): boolean {
    return (
      serializeMarkdownTable(other.table) === serializeMarkdownTable(this.table) &&
      other.tableClass === this.tableClass &&
      cellKey(other.selected?.cell ?? null) === cellKey(this.selected?.cell ?? null) &&
      other.selected?.editing === this.selected?.editing &&
      other.selected?.draft === this.selected?.draft &&
      other.canEdit === this.canEdit
    );
  }

  toDOM(view: EditorView): HTMLElement {
    const wrapper = document.createElement('div');
    wrapper.className = `ge-table-widget ${this.tableClass}${this.selected ? ' ge-table-selected' : ''}`;
    if (this.canEdit && this.selected) wrapper.append(this.renderControls(view));

    const scroll = document.createElement('div');
    scroll.className = 'ge-table-scroll';
    const table = document.createElement('table');
    table.className = 'ge-table-rendered';
    table.append(this.renderHead(view), this.renderBody(view));
    scroll.append(table);
    wrapper.append(scroll);
    return wrapper;
  }

  ignoreEvent(): boolean {
    return false;
  }

  private select(view: EditorView, event: MouseEvent, cell: TableCellRef): void {
    event.preventDefault();
    event.stopPropagation();
    if (!this.canEdit || view.state.readOnly) return;
    if (event.ctrlKey || event.metaKey) {
      revealTableSource(view, cell);
      view.dispatch({ effects: selectTableCell.of(null) });
      return;
    }
    const modelCell = tableCell(this.table, cell);
    view.dispatch({
      selection: EditorSelection.cursor(modelCell?.sourceFrom ?? this.table.from),
      effects: selectTableCell.of({ tableFrom: this.table.from, tableTo: this.table.to, cell, editing: false, draft: null }),
    });
  }

  private startEdit(view: EditorView, cell: TableCellRef, seed?: string): void {
    const modelCell = tableCell(this.table, cell);
    if (!modelCell || !this.canEdit || view.state.readOnly) return;
    view.dispatch({ effects: selectTableCell.of({ tableFrom: this.table.from, tableTo: this.table.to, cell, editing: true, draft: seed ?? modelCell.text }) });
  }
}
```

```ts
function buildTableDecorations(view: EditorView, tableClass: string, preview: boolean, canEdit: boolean): DecorationSet {
  const widgets: Range<Decoration>[] = [];
  const selected = view.state.field(selectedTableCellField);
  syntaxTree(view.state).iterate({
    enter(node) {
      if (node.name !== 'Table') return;

      if (!parsed) return;
      if (!preview && canEdit && view.state.selection.main.from >= node.from && view.state.selection.main.to <= node.to && !selected) return;

        widget: new TableWidget(parsed, tableClass, selected?.tableFrom === node.from ? selected : null, canEdit && !preview),
        block: true,
      }).range(node.from, node.to));
    },
  });
  return Decoration.set(widgets, true);
}
```

```ts
function renderCellInlineMarkdown(cell: HTMLElement, source: string): void {
  const codePattern = /`([^`]+)`/g;
  const boldPattern = /\*\*([^*]+)\*\*/g;
  const linkPattern = /\[([^\]]+)\]\(([^)]+)\)/g;
  let html = escapeHtml(source);
  html = html.replace(codePattern, '<code class="ge-table-cell-code">$1</code>');
  html = html.replace(boldPattern, '<strong class="ge-table-cell-bold">$1</strong>');
  html = html.replace(linkPattern, '<a class="ge-table-cell-link" href="$2">$1</a>');
  cell.innerHTML = html;
}

function escapeHtml(value: string): string {
  return value
    .replaceAll('&', '&amp;')
    .replaceAll('<', '&lt;')
    .replaceAll('>', '&gt;')
    .replaceAll('"', '&quot;');
}
```
